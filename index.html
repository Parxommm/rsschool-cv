RGB To Hex Conversion

            function rgb(r, g, b) {
              const toHex = (num) => {
               if (num <= 0) {
                 num = '00';
               } else if (num > 255) {
                 num = 'FF';
               } else {
                 num = num.toString(16).toUpperCase();
               }
               if (num.length === 1) {
                 num = 0 + num;
               }
               return num;
              }
              r = toHex(r);
              g = toHex(g);
              b = toHex(b);
              
              return r + g + b;
             }
